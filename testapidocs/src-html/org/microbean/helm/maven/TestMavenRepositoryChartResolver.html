120  @Before
121  public void setup() {
122    assertNotNull(settings);
123    
124    final DefaultRepositorySystemSession repositorySystemSession = MavenRepositorySystemUtils.newSession();
125    assertNotNull(repositorySystemSession);
126    // repositorySystemSession.setUpdatePolicy("always");
127    repositorySystemSession.setTransferListener(new TransferListener());
128    repositorySystemSession.setOffline(settings.isOffline());
129    repositorySystemSession.setCache(new DefaultRepositoryCache());
130
131    final Collection<? extends Mirror> mirrors = settings.getMirrors();
132    if (mirrors != null && !mirrors.isEmpty()) {
133      final DefaultMirrorSelector mirrorSelector = new DefaultMirrorSelector();
134      for (final Mirror mirror : mirrors) {
135        assert mirror != null;
136        mirrorSelector.add(mirror.getId(),
137                           mirror.getUrl(),
138                           mirror.getLayout(),
139                           false, /* not a repository manager; settings.xml does not encode this information */
140                           mirror.getMirrorOf(),
141                           mirror.getMirrorOfLayouts());
142      }
143      repositorySystemSession.setMirrorSelector(mirrorSelector);
144    }
145
146    String localRepositoryString = settings.getLocalRepository();
147    if (localRepositoryString == null) {
148      localRepositoryString = System.getProperty("user.home") + "/.m2/repository";
149    }
150    
151    final LocalRepository localRepository = new LocalRepository(localRepositoryString);
152    final LocalRepositoryManager localRepositoryManager = repositorySystem.newLocalRepositoryManager(repositorySystemSession, localRepository);
153    assertNotNull(localRepositoryManager);    
154    repositorySystemSession.setLocalRepositoryManager(localRepositoryManager);
155
156     List<RemoteRepository> remoteRepositories = new ArrayList<>();
157    final Map<String, Profile> profiles = settings.getProfilesAsMap();
158    if (profiles != null && !profiles.isEmpty()) {
159      final Collection<String> activeProfileKeys = settings.getActiveProfiles();
160      if (activeProfileKeys != null && !activeProfileKeys.isEmpty()) {
161        for (final String activeProfileKey : activeProfileKeys) {
162          final Profile activeProfile = profiles.get(activeProfileKey);
163          if (activeProfile != null) {
164            final Collection<Repository> repositories = activeProfile.getRepositories();
165            if (repositories != null && !repositories.isEmpty()) {
166              for (final Repository repository : repositories) {
167                if (repository != null) {
168                  Builder builder = new Builder(repository.getId(), repository.getLayout(), repository.getUrl());
169
170                  final org.apache.maven.settings.RepositoryPolicy settingsReleasePolicy = repository.getReleases();
171                  if (settingsReleasePolicy != null) {
172                    final org.eclipse.aether.repository.RepositoryPolicy releasePolicy = new org.eclipse.aether.repository.RepositoryPolicy(settingsReleasePolicy.isEnabled(), settingsReleasePolicy.getUpdatePolicy(), settingsReleasePolicy.getChecksumPolicy());
173                    builder = builder.setReleasePolicy(releasePolicy);
174                  }
175
176                  final org.apache.maven.settings.RepositoryPolicy settingsSnapshotPolicy = repository.getSnapshots();
177                  if (settingsSnapshotPolicy != null) {
178                    final org.eclipse.aether.repository.RepositoryPolicy snapshotPolicy = new org.eclipse.aether.repository.RepositoryPolicy(settingsSnapshotPolicy.isEnabled(), settingsSnapshotPolicy.getUpdatePolicy(), settingsSnapshotPolicy.getChecksumPolicy());
179                    builder = builder.setSnapshotPolicy(snapshotPolicy);
180                  }
181                  
182                  final RemoteRepository remoteRepository = builder.build();
183                  assert remoteRepository != null;
184                  remoteRepositories.add(remoteRepository);
185                }
186              }
187            }
188          }
189        }
190      }
191    }
192    final RemoteRepository mavenCentral = new Builder("central", "default", "http://central.maven.org/maven2/").build();
193    assert mavenCentral != null;
194    remoteRepositories.add(mavenCentral);
195    remoteRepositories = repositorySystem.newResolutionRepositories(repositorySystemSession, remoteRepositories);
196    assertNotNull(remoteRepositories);
197
198    this.resolver = new MavenRepositoryChartResolver(repositorySystem, repositorySystemSession, remoteRepositories);
199  }

210  private static final Settings getSettings() throws SettingsBuildingException {
211    final SettingsBuilder settingsBuilder = new DefaultSettingsBuilderFactory().newInstance(); // this method should be static!
212    assert settingsBuilder != null;
213    final DefaultSettingsBuildingRequest settingsBuildingRequest = new DefaultSettingsBuildingRequest();
214    settingsBuildingRequest.setSystemProperties(System.getProperties());
215    // settingsBuildingRequest.setUserProperties(userProperties); // TODO: implement this
216    settingsBuildingRequest.setGlobalSettingsFile(new File("/usr/local/maven/conf/settings.xml")); // TODO: do this for real
217    settingsBuildingRequest.setUserSettingsFile(new File(new File(System.getProperty("user.home")), ".m2/settings.xml"));
218    final SettingsBuildingResult settingsBuildingResult = settingsBuilder.build(settingsBuildingRequest);
219    assert settingsBuildingResult != null;
220    final List<SettingsProblem> settingsBuildingProblems = settingsBuildingResult.getProblems();
221    if (settingsBuildingProblems != null && !settingsBuildingProblems.isEmpty()) {
222      throw new SettingsBuildingException(settingsBuildingProblems);
223    }
224    return settingsBuildingResult.getEffectiveSettings();
225  }
